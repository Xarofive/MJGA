<h2 style="color: #34495e; margin-top: 30px;">1. Что возвращает for range при итерации по строке</h2>

<p>При использовании <code>for range</code> со строкой Go автоматически декодирует UTF-8 и возвращает два значения: <strong>байтовый индекс</strong> начала текущего символа и саму <strong>руну</strong> (копию символа). Это ключевая особенность, которую необходимо понимать для корректной работы со строками.</p>

<pre style="background: #f8f9fa; border: 1px solid #e9ecef; border-radius: 5px; padding: 15px; margin: 10px 0; overflow-x: auto;">
<code class="language-go">package main

import "fmt"

func main() {
    // Демонстрация того, что возвращает for range
    s := "Go!"

    fmt.Println("=== Анатомия for range ===")
    fmt.Printf("Строка: %q, длина: %d байт\n\n", s, len(s))

    // Полная форма с двумя переменными
    fmt.Println("Полная форма: for i, r := range s")
    for i, r := range s {
        fmt.Printf("  i=%d (байтовый индекс), r=%c (руна U+%04X)\n", i, r, r)
    }

    // Только индексы
    fmt.Println("\nТолько индексы: for i := range s")
    for i := range s {
        fmt.Printf("  i=%d, s[i]=%d (байт, НЕ руна!)\n", i, s[i])
    }

    // Только руны
    fmt.Println("\nТолько руны: for _, r := range s")
    for _, r := range s {
        fmt.Printf("  r=%c\n", r)
    }

    // ВАЖНО: r - это копия, не ссылка
    fmt.Println("\n=== r - это копия ===")
    original := "ABC"
    for i, r := range original {
        r = 'X' // Это не изменит строку!
        fmt.Printf("После r='X': original[%d] = %c, r = %c\n", i, original[i], r)
    }
    fmt.Printf("Исходная строка не изменилась: %q\n", original)
}</code></pre>

<h2 style="color: #34495e; margin-top: 30px;">2. Почему индексы &quot;скачут&quot;: демонстрация на разных символах</h2>

<p>Главная особенность <code>for range</code> по строке &mdash; индекс i указывает на байтовую позицию начала символа, а не на порядковый номер символа. Для многобайтовых символов индексы будут &quot;скакать&quot; в зависимости от размера символа в байтах.</p>

<pre style="background: #f8f9fa; border: 1px solid #e9ecef; border-radius: 5px; padding: 15px; margin: 10px 0; overflow-x: auto;">
<code class="language-go">package main

import (
    "fmt"
    "unicode/utf8"
)

func main() {
    // Три тестовые строки: ASCII, кириллица, эмодзи
    ascii := "Hello"
    cyrillic := "Привет"
    mixed := "Hi🙂!"

    fmt.Println("=== ASCII строка (1 байт на символ) ===")
    fmt.Printf("Строка: %q\n", ascii)
    fmt.Printf("Байтов: %d, Символов: %d\n", len(ascii), utf8.RuneCountInString(ascii))

    symbolNum := 0
    for i, r := range ascii {
        fmt.Printf("Символ №%d: байт_индекс=%d, руна=%c\n", symbolNum, i, r)
        symbolNum++
    }

    fmt.Println("\n=== Кириллица (2 байта на символ) ===")
    fmt.Printf("Строка: %q\n", cyrillic)
    fmt.Printf("Байтов: %d, Символов: %d\n", len(cyrillic), utf8.RuneCountInString(cyrillic))

    symbolNum = 0
    for i, r := range cyrillic {
        fmt.Printf("Символ №%d: байт_индекс=%d, руна=%c (скачок на %d)\n",
            symbolNum, i, r, utf8.RuneLen(r))
        symbolNum++
    }

    fmt.Println("\n=== Смешанная строка ===")
    fmt.Printf("Строка: %q\n", mixed)
    fmt.Printf("Байтов: %d, Символов: %d\n", len(mixed), utf8.RuneCountInString(mixed))

    symbolNum = 0
    prevIndex := -1
    for i, r := range mixed {
        jump := ""
        if prevIndex &gt;= 0 {
            jump = fmt.Sprintf(" (скачок +%d)", i-prevIndex)
        }
        fmt.Printf("Символ №%d: байт_индекс=%d, руна=%c, размер=%d байт%s\n",
            symbolNum, i, r, utf8.RuneLen(r), jump)
        prevIndex = i
        symbolNum++
    }

    // Визуализация байтовой структуры
    fmt.Println("\n=== Байтовая карта строки \"Hi🙂!\" ===")
    fmt.Println("Индекс: 0  1  2  3  4  5  6")
    fmt.Println("Байт:   H  i  [----🙂----]  !")
    fmt.Println("        ↑  ↑  ↑           ↑")
    fmt.Println("range:  0  1  2           6")
}</code></pre>

<h2 style="color: #34495e; margin-top: 30px;">3. Практическое применение: безопасное срезание строк</h2>

<div style="background: #e3f2fd; border: 1px solid #90caf9; border-radius: 8px; padding: 15px; margin: 20px 0;">
<p style="margin: 0;"><strong>📝 Напоминание:</strong> Разница между <code>len(s)</code> (байты) и <code>utf8.RuneCountInString(s)</code> (символы) подробно разобрана в предыдущем уроке. Здесь покажем, как знание индексов из <code>for range</code> помогает безопасно работать с подстроками.</p>
</div>

<p>Зная, что индексы в <code>for range</code> &mdash; это байтовые позиции, становится понятно, почему <code>s[:6]</code> для кириллической строки может дать неожиданный результат. Для безопасного извлечения подстроки преобразуйте строку в <code>[]rune</code>:</p>

<pre style="background: #f8f9fa; border: 1px solid #e9ecef; border-radius: 5px; padding: 15px; margin: 10px 0; overflow-x: auto;">
<code class="language-go">s := "Привет, мир! 🌍"

// НЕПРАВИЛЬНО: s[:6] берёт 6 БАЙТОВ, а не 6 символов
wrongSlice := s[:6]  // "При" (3 кириллических символа × 2 байта)

// ПРАВИЛЬНО: через преобразование в руны с проверкой границ
runes := []rune(s)
n := 6
if n &gt; len(runes) {
    n = len(runes)
}
rightSlice := string(runes[:n])  // "Привет" (6 символов)</code></pre>

<div style="background: #fee2e2; border: 1px solid #ef4444; border-radius: 8px; padding: 15px; margin: 20px 0;">
<p style="margin: 0;"><strong>⚠️ Правило:</strong> Если вам нужны первые N <strong>символов</strong> &mdash; используйте <code>[]rune(s)[:N]</code>, но всегда проверяйте, что N не превышает <code>len(runes)</code>, иначе возникнет ошибка времени выполнения.</p>
</div>

<h2 style="color: #34495e; margin-top: 30px;">4. Ручной проход с utf8.DecodeRuneInString</h2>

<p>Иногда необходим более точный контроль над итерацией, например, для обработки ошибок декодирования или оптимизации. В таких случаях используется <code>utf8.DecodeRuneInString</code> для ручного прохода по строке.</p>

<pre style="background: #f8f9fa; border: 1px solid #e9ecef; border-radius: 5px; padding: 15px; margin: 10px 0; overflow-x: auto;">
<code class="language-go">package main

import (
    "fmt"
    "unicode/utf8"
)

// findFirstInvalid возвращает индекс первого невалидного UTF-8 байта
// в строке или -1, если вся строка — корректный UTF-8.
func findFirstInvalid(s string) int {
    for i := 0; i &lt; len(s); {
        r, size := utf8.DecodeRuneInString(s[i:])
        if r == utf8.RuneError &amp;&amp; size == 1 {
            return i
        }
        i += size
    }
    return -1
}

// skipRunes пропускает первые n символов (рун) строки и возвращает остаток.
func skipRunes(s string, n int) string {
    pos := 0
    for i := 0; i &lt; n &amp;&amp; pos &lt; len(s); i++ {
        _, size := utf8.DecodeRuneInString(s[pos:])
        pos += size
    }
    return s[pos:]
}

func main() {
    s := "Hello, 世界! 🚀"

    fmt.Printf("Строка: %q\n", s)
    fmt.Printf("Байтов: %d, Символов: %d\n\n",
        len(s), utf8.RuneCountInString(s))

    // Сравнение: for range vs ручной проход
    fmt.Println("=== for range (автоматический) ===")
    for i, r := range s {
        fmt.Printf("Позиция %2d: %c (U+%04X)\n", i, r, r)
    }

    fmt.Println("\n=== Ручной проход с DecodeRuneInString ===")
    for i := 0; i &lt; len(s); {
        r, size := utf8.DecodeRuneInString(s[i:])
        fmt.Printf("Позиция %2d: %c (U+%04X), размер %d байт\n", i, r, r, size)
        i += size
    }

    // Обработка невалидного UTF-8
    fmt.Println("\n=== Обработка невалидного UTF-8 ===")

    // Строка с невалидными байтами
    invalid := "Hello" + string([]byte{0xFF, 0xFE}) + "World"
    fmt.Printf("Строка с невалидными байтами: %q\n", invalid)

    fmt.Println("\nfor range (заменяет невалидные на U+FFFD):")
    for i, r := range invalid {
        if r == utf8.RuneError {
            fmt.Printf("  Позиция %d: ОШИБКА! (U+FFFD)\n", i)
        } else {
            fmt.Printf("  Позиция %d: %c\n", i, r)
        }
    }

    fmt.Println("\nРучной проход с детекцией ошибок:")
    for i := 0; i &lt; len(invalid); {
        r, size := utf8.DecodeRuneInString(invalid[i:])
        if r == utf8.RuneError &amp;&amp; size == 1 {
            fmt.Printf("  Позиция %d: Невалидный байт 0x%02X\n", i, invalid[i])
        } else {
            fmt.Printf("  Позиция %d: %c (размер %d)\n", i, r, size)
        }
        i += size
    }

    if pos := findFirstInvalid(invalid); pos &gt;= 0 {
        fmt.Printf("\nПервый невалидный байт найден на позиции %d\n", pos)
    }

    // Преимущества ручного прохода
    fmt.Println("\n=== Преимущества ручного прохода ===")

    // Подсчёт только ASCII символов
    asciiCount := 0
    for i := 0; i &lt; len(s); {
        r, size := utf8.DecodeRuneInString(s[i:])
        if r &lt; 128 {
            asciiCount++
        }
        i += size
    }
    fmt.Printf("ASCII символов в %q: %d\n", s, asciiCount)

    fmt.Printf("Пропустить 7 символов: %q\n", skipRunes(s, 7))
}</code></pre>

<code class="language-go">package main

import "fmt"

func FindRunePosition(s string, target rune) int {
    position := 0
    for _, r := range s {
        if r == target {
            return position
        }
        position++
    }
    return -1
}

func FindRunePositionWithByte(s string, target rune) (int, int) {
    runePos := 0
    for byteIdx, r := range s {
        if r == target {
            return runePos, byteIdx
        }
        runePos++
    }
    return -1, -1
}

func main() {
    s := "Hello, мир! 🌍"

    tests := []rune{'H', 'o', 'м', '!', '🌍', 'x'}

    fmt.Printf("Строка: %q\n\n", s)

    for _, r := range tests {
        pos := FindRunePosition(s, r)
        if pos &gt;= 0 {
            fmt.Printf("Руна %c найдена на позиции %d\n", r, pos)
        } else {
            fmt.Printf("Руна %c не найдена\n", r)
        }

    fmt.Println("\n=== Позиция символа vs байтовый индекс ===")
    for _, r := range tests {
        runePos, bytePos := FindRunePositionWithByte(s, r)
        if runePos &gt;= 0 {
            fmt.Printf("Руна %c: символ №%d, байт №%d\n", r, runePos, bytePos)

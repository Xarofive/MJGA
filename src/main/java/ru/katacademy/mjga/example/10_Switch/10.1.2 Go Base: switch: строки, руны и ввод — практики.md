<h2 style="color: #34495e; margin-top: 28px;">1. Классификация руны</h2>

<div style="background: #fff8e1; border: 1px solid #ffc107; border-radius: 8px; padding: 14px; margin: 16px 0;">
<h4 style="color: #f57c00; margin-top: 0;">📚 Теория: switch без выражения для сложных проверок</h4>

<p style="margin-bottom: 10px;">При классификации символов часто нужно проверить несколько взаимоисключающих условий. Switch без выражения читается лучше цепочки if-else и явно показывает, что проверки взаимоисключающие.</p>

<p style="margin: 0;">Пакет unicode предоставляет функции для проверки категорий символов согласно стандарту Unicode. Это работает для всех языков, не только ASCII.</p>
</div>

<div style="background: #f0f4f8; border-radius: 8px; padding: 16px; margin: 16px 0;">

<code class="language-go">package main

import (
"fmt"
"unicode"
)

func classifyRune(r rune) string {
switch {
case unicode.IsLetter(r):
return "буква"
case unicode.IsDigit(r):
return "цифра"
case unicode.IsSpace(r):
return "пробельный символ"
case unicode.IsPunct(r):
return "пунктуация"
case unicode.IsSymbol(r):
return "символ"
case unicode.IsControl(r):
return "управляющий символ"
default:
return "другое"
}
}

func main() {
testRunes := []rune{'A', 'б', '5', ' ', '.', '€', '\n', '😀'}

    for _, r := range testRunes {
        classification := classifyRune(r)
        if unicode.IsPrint(r) {
            fmt.Printf("%c → %s\n", r, classification)
        } else {
            fmt.Printf("U+%04X → %s\n", r, classification)
        }
    }
}</code></pre>

<h2 style="color: #34495e; margin-top: 28px;">2. Подсчёт гласных в латинице</h2>

<div style="background: #e3f2fd; border: 1px solid #90caf9; border-radius: 8px; padding: 14px; margin: 16px 0;">
<h4 style="color: #1565c0; margin-top: 0;">💡 Важно: руны vs байты</h4>

<p style="margin-bottom: 10px;">При работе со строками важно различать:</p>

<ul style="margin: 10px 0;">
	<li><code>len(s)</code> &mdash; количество байтов в строке</li>
	<li><code>utf8.RuneCountInString(s)</code> &mdash; количество рун (символов)</li>
	<li><code>range s</code> &mdash; итерация по рунам, а не байтам</li>
</ul>

<p style="margin: 0;">Для корректной работы с многобайтовыми символами всегда используйте итерацию через range или явное декодирование рун.</p>
</div>

<div style="background: #f0f4f8; border-radius: 8px; padding: 16px; margin: 16px 0;">

<code class="language-go">package main

import (
"fmt"
"unicode"
"unicode/utf8"
)

func countVowelsConsonants(s string) (int, int) {
vowels := 0
consonants := 0

    for _, r := range s {
        if !unicode.IsLetter(r) {
            continue
        }

        // Приводим к нижнему регистру для единообразной проверки
        lower := unicode.ToLower(r)

        // Считаем только латинские гласные;
        // буквы других алфавитов попадают в счётчик согласных
        switch lower {
        case 'a', 'e', 'i', 'o', 'u':
            vowels++
        default:
            consonants++
        }
    }
    return vowels, consonants
}

func main() {
texts := []string{
"Hello World",
"Привет Мир",
"Café ☕",
"GO is awesome!",
}

    for _, text := range texts {
        v, c := countVowelsConsonants(text)
        fmt.Printf("%-20s: %d гласных, %d согласных (байт: %d, рун: %d)\n",
            text, v, c, len(text), utf8.RuneCountInString(text))
    }
}</code></pre>

<h2 style="color: #34495e; margin-top: 28px;">3. Маршрутизация команд</h2>

<div style="background: #fff8e1; border: 1px solid #ffc107; border-radius: 8px; padding: 14px; margin: 16px 0;">
<h4 style="color: #f57c00; margin-top: 0;">📚 Паттерн: Command Router</h4>

<p style="margin-bottom: 10px;">Типичный паттерн для интерактивных программ:</p>

<ol style="margin: 10px 0;">
	<li>Читаем строку через Scanner</li>
	<li>Нормализуем ввод (обрезаем пробелы, приводим к нижнему регистру)</li>
	<li>Разбиваем на слова через strings.Fields</li>
	<li>Маршрутизируем первое слово через switch</li>
	<li>Обрабатываем параметры команды</li>
</ol>

<p style="margin: 0;"><strong>Важно:</strong> всегда проверяйте scanner.Err() после завершения цикла для обнаружения ошибок чтения.</p>
</div>

<div style="background: #f0f4f8; border-radius: 8px; padding: 16px; margin: 16px 0;">
<pre style="background: #263238; color: #aed581; padding: 12px; border-radius: 4px; margin: 10px 0;">
<code class="language-go">package main

import (
"bufio"
"fmt"
"strings"
)

func processCommands(input string) error {
scanner := bufio.NewScanner(strings.NewReader(input))
lineNum := 0

    fmt.Println("Введите команды (quit для выхода):")

    for scanner.Scan() {
        line := strings.TrimSpace(scanner.Text())

        // Пропускаем пустые строки
        if line == "" {
            continue
        }

        lineNum++ // Считаем только непустые строки

        // Разбиваем на слова
        parts := strings.Fields(line)
        if len(parts) == 0 {
            continue
        }

        // Первое слово — команда (нормализуем регистр)
        cmd := strings.ToLower(parts[0])
        args := parts[1:]

        switch cmd {
        case "add":
            if len(args) == 0 {
                fmt.Println("Ошибка: add требует аргументы")
            } else {
                fmt.Printf("Добавляем: %s\n", strings.Join(args, " "))
            }

        case "list", "ls":
            fmt.Println("Список элементов:")
            fmt.Println("  - элемент 1")
            fmt.Println("  - элемент 2")

        case "delete", "del", "rm":
            if len(args) == 0 {
                fmt.Println("Ошибка: delete требует ID элемента")
            } else {
                fmt.Printf("Удаляем элемент: %s\n", args[0])
            }

        case "help", "?":
            fmt.Println("Доступные команды:")
            fmt.Println("  add &lt;текст&gt;  - добавить элемент")
            fmt.Println("  list         - показать список")
            fmt.Println("  delete &lt;id&gt;  - удалить элемент")
            fmt.Println("  quit         - выйти")

        case "quit", "exit", "q":
            fmt.Println("Завершение работы...")
            return nil

        default:
            fmt.Printf("Неизвестная команда: %q (строка %d)\n", cmd, lineNum)
            fmt.Println("Введите 'help' для справки")
        }
    }

    // ВАЖНО: проверяем ошибки после цикла
    if err := scanner.Err(); err != nil {
        return scanner.Err()
    }

    // EOF — это нормальное завершение
    fmt.Println("\nКонец ввода (EOF)")
    return nil
}</code></pre>

<p style="margin: 12px 0;">Функция принимает входной текст в виде строки. Внутри мы оборачиваем строку в <code>strings.NewReader</code> и передаём в <code>bufio.NewScanner</code> &mdash; это позволяет читать данные построчно тем же сканером, что и при чтении из стандартного ввода. В практике mini-REPL вы будете писать функцию, читающую команды и печатающую результат &mdash; детали сигнатуры см. в описании задачи.</p>

<p style="margin: 12px 0;">Вот как вызвать функцию <code>processCommands</code>:</p>

<pre style="background: #263238; color: #aed581; padding: 12px; border-radius: 4px; margin: 10px 0;">
<code class="language-go">func main() {
    input := `ADD первая задача
list
Delete 1
unknown command
help
quit`

    if err := processCommands(input); err != nil {
        fmt.Printf("Ошибка: %v\n", err)
    }
}</code></pre>

<h3 style="color: #455a64; margin-top: 24px;">Пример 3: Интерактивный калькулятор</h3>

<div style="background: #f0f4f8; border-radius: 8px; padding: 16px; margin: 16px 0;">

<code class="language-go">package main

import (
"bufio"
"fmt"
"strconv"
"strings"
)

func calculate(input string) {
scanner := bufio.NewScanner(strings.NewReader(input))

    fmt.Println("Калькулятор (формат: число операция число)")

    for scanner.Scan() {
        line := strings.TrimSpace(scanner.Text())
        if line == "" {
            continue
        }

        parts := strings.Fields(line)
        if len(parts) != 3 {
            if strings.ToLower(line) == "quit" {
                break
            }
            fmt.Println("Формат: число операция число")
            continue
        }

        a, err1 := strconv.ParseFloat(parts[0], 64)
        b, err2 := strconv.ParseFloat(parts[2], 64)

        if err1 != nil || err2 != nil {
            fmt.Println("Ошибка: введите числа")
            continue
        }

        op := parts[1]
        var result float64
        var valid = true

        switch op {
        case "+":
            result = a + b
        case "-":
            result = a - b
        case "*", "×":
            result = a * b
        case "/", "÷":
            if b == 0 {
                fmt.Println("Ошибка: деление на ноль")
                continue
            }
            result = a / b
        default:
            fmt.Printf("Неизвестная операция: %s\n", op)
            valid = false
        }

        if valid {
            fmt.Printf("%.2f %s %.2f = %.2f\n", a, op, b, result)
        }
    }

    if err := scanner.Err(); err != nil {
        fmt.Printf("Ошибка чтения: %v\n", err)
    }
}

func main() {
input := `10 + 20
5.5 * 2
100 / 4
7 - 3.5
quit`

    calculate(input)

<h3 style="color: #455a64; margin-top: 24px;">Пример 4: Корректная проверка ошибок Scanner</h3>

<div style="background: #f0f4f8; border-radius: 8px; padding: 16px; margin: 16px 0;">

<code class="language-go">package main

import (
"bufio"
"fmt"
"strings"
)

func readAndProcess(data string) error {
scanner := bufio.NewScanner(strings.NewReader(data))
processed := 0

    for scanner.Scan() {
        line := scanner.Text()

        switch {
        case strings.HasPrefix(line, "#"):
            // Комментарий — пропускаем
            continue
        case strings.TrimSpace(line) == "":
            // Пустая строка — пропускаем
            continue
        case strings.ToLower(line) == "stop":
            fmt.Printf("Остановка. Обработано строк: %d\n", processed)
            return nil
        default:
            fmt.Printf("Обработка: %s\n", line)
            processed++
        }
    }

    // КРИТИЧНО: проверяем ошибки после цикла!
    if err := scanner.Err(); err != nil {
        return fmt.Errorf("ошибка сканирования после %d строк: %v", processed, err)
    }

    fmt.Printf("EOF достигнут. Всего обработано: %d\n", processed)
    return nil
}

func main() {
data := `# Это комментарий
первая строка

вторая строка
# ещё комментарий
третья строка
stop
эта строка не будет обработана`

    if err := readAndProcess(data); err != nil {
        fmt.Printf("Ошибка: %v\n", err)
    }
}</code></pre>

<h3 style="color: #455a64; margin-top: 24px;">Пример 5: Разница между байтами и рунами</h3>

<code class="language-go">package main

import (

func demonstrateRuneVsByte() {

    for _, s := range samples {

            case r &lt; 128:

                fmt.Printf(" (начинается с BMP)")
            default:
                fmt.Printf(" (начинается вне BMP)")
            }
            break // только первая руна
        }
        fmt.Println()
    }
}

func main() {
demonstrateRuneVsByte()

    // Пример обхода строки побайтно vs по рунам
    s := "Go🚀"

    fmt.Printf("\nПобайтный обход %q:\n", s)
    for i := 0; i &lt; len(s); i++ {
        fmt.Printf("  [%d] = 0x%02X\n", i, s[i])
    }

    fmt.Printf("\nОбход по рунам %q:\n", s)
    for i, r := range s {
        fmt.Printf("  [%d] = %c (U+%04X)\n", i, r, r)
    }
}</code></pre>

<div style="background: #e8f5e9; border: 1px solid #66bb6a; border-radius: 4px; padding: 10px; margin-top: 10px;"><strong>📋 Ожидаемый вывод:</strong>

<pre style="margin: 5px 0; color: #2e7d32;">

  [2] = 0xF0

<div style="background: #f1f8e9; border: 1px solid #7cb342; border-radius: 8px; padding: 12px; margin: 12px 0;">
<h4 style="color: #558b2f; margin-top: 0;">💡 Подсказка: Ментальная модель</h4>

<p style="margin: 10px 0;">Представьте switch для текста как почтовый сортировщик: каждая входящая команда (конверт) проверяется по адресу (первое слово), и направляется в нужный отдел (case). Неизвестные адреса попадают в отдел возврата (default).</p>

<p style="margin: 0;">При работе с рунами думайте о них как о &laquo;логических символах&raquo;, а не байтах. Часто один эмодзи представлен одной руной и занимает 4 байта, но встречаются последовательности из нескольких рун (флаги стран, модификаторы тона кожи). Функции unicode.* работают с отдельными рунами.</p>

<h2 style="color: #34495e; margin-top: 28px;">5. Типичные ошибки</h2>

<div style="background: #ffebee; border: 1px solid #ef5350; border-radius: 8px; padding: 14px; margin: 16px 0;">
<h4 style="color: #c62828; margin-top: 0;">❌ Ошибка 1: Забыли нормализовать регистр</h4>

<pre style="background: #263238; color: #aed581; padding: 12px; border-radius: 4px; margin: 10px 0;">
<code class="language-go">package main

import "unicode"

// ПЛОХО: не учитывает заглавные буквы
func isVowelBad(r rune) bool {
    switch r {
    case 'a', 'e', 'i', 'o', 'u':
        return true
    }
    return false // isVowelBad('A') вернёт false!
}

// ХОРОШО: нормализуем регистр через unicode.ToLower
func isVowelGood(r rune) bool {
    switch unicode.ToLower(r) {
    case 'a', 'e', 'i', 'o', 'u':
        return true
    }
    return false // isVowelGood('A') вернёт true
}</code></pre>

<p style="margin: 0;"><strong>Решение:</strong> используйте unicode.ToLower перед проверкой, чтобы корректно обрабатывать любой регистр.</p>
</div>

<div style="background: #ffebee; border: 1px solid #ef5350; border-radius: 8px; padding: 14px; margin: 16px 0;">
<h4 style="color: #c62828; margin-top: 0;">❌ Ошибка 2: Отсутствие нормализации ввода</h4>

<pre style="background: #263238; color: #aed581; padding: 12px; border-radius: 4px; margin: 10px 0;">
<code class="language-go">package main

import "strings"

// ПЛОХО: не обрабатывает вариации
func processBad(command string) {
    switch command {
    case "quit":
        return
    }
}

// ХОРОШО: нормализация перед проверкой
func processGood(command string) {
    cmd := strings.ToLower(strings.TrimSpace(command))
    switch cmd {
    case "quit", "exit", "q":
        return
    }
}</code></pre>

<p style="margin: 0;"><strong>Решение:</strong> всегда нормализуйте пользовательский ввод перед маршрутизацией.</p>
</div>

<div style="background: #ffebee; border: 1px solid #ef5350; border-radius: 8px; padding: 14px; margin: 16px 0;">
<h4 style="color: #c62828; margin-top: 0;">❌ Ошибка 3: Игнорирование scanner.Err()</h4>

<pre style="background: #263238; color: #aed581; padding: 12px; border-radius: 4px; margin: 10px 0;">
<code class="language-go">// ПЛОХО: ошибки теряются
scanner := bufio.NewScanner(reader)
for scanner.Scan() {
    fmt.Println(scanner.Text())
}
// Что если была ошибка чтения? Мы никогда не узнаем!

// ХОРОШО: проверка после цикла
scanner := bufio.NewScanner(reader)
for scanner.Scan() {
    fmt.Println(scanner.Text())
}
if err := scanner.Err(); err != nil {
    fmt.Println("ошибка чтения:", err)
}</code></pre>

<p style="margin: 0;"><strong>Решение:</strong> всегда проверяйте <code>scanner.Err()</code> после завершения цикла сканирования.</p>
</div>

<div style="background: #ffebee; border: 1px solid #ef5350; border-radius: 8px; padding: 14px; margin: 16px 0;">
<h4 style="color: #c62828; margin-top: 0;">❌ Ошибка 4: Путаница байтов и рун</h4>

<pre style="background: #263238; color: #aed581; padding: 12px; border-radius: 4px; margin: 10px 0;">
<code class="language-go">package main

import "fmt"

func demonstrateError() {
    // ПЛОХО: обход по байтам для Unicode-строки
    s := "Привет"
    for i := 0; i &lt; len(s); i++ {
        fmt.Printf("%c", s[i]) // Мусор для кириллицы!
    }

    fmt.Println()

    // ХОРОШО: обход по рунам
    for _, r := range s {
        fmt.Printf("%c", r)
    }
}</code></pre>

<p style="margin: 0;"><strong>Решение:</strong> используйте range для итерации по рунам, len() показывает байты, а не символы.</p>
</div>
